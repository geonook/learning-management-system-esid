## SQL Schema

```sql
CREATE TABLE map_achievement_norms_2025 (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  level TEXT NOT NULL CHECK (level IN ('student', 'school')),
  subject TEXT NOT NULL CHECK (subject IN ('Mathematics', 'Reading', 'Language Usage', 'Science')),
  grade INTEGER NOT NULL,
  season TEXT NOT NULL CHECK (season IN ('fall', 'winter', 'spring')),
  mean_rit DECIMAL(6,2) NOT NULL,
  sd DECIMAL(5,2) NOT NULL,
  
  UNIQUE(level, subject, grade, season)
);

-- Index for common lookups
CREATE INDEX idx_norms_lookup ON map_achievement_norms_2025(subject, grade, season, level);
```

## Percentile Calculation Function

```sql
-- PostgreSQL function to calculate achievement percentile
CREATE OR REPLACE FUNCTION calculate_achievement_percentile(
  p_rit_score INTEGER,
  p_subject TEXT,
  p_grade INTEGER,
  p_season TEXT,
  p_level TEXT DEFAULT 'student'
) RETURNS INTEGER AS $$
DECLARE
  v_mean DECIMAL;
  v_sd DECIMAL;
  v_z DECIMAL;
  v_percentile INTEGER;
BEGIN
  SELECT mean_rit, sd INTO v_mean, v_sd
  FROM map_achievement_norms_2025
  WHERE subject = p_subject
    AND grade = p_grade
    AND season = p_season
    AND level = p_level;
    
  IF v_mean IS NULL THEN
    RETURN NULL;
  END IF;
  
  v_z := (p_rit_score - v_mean) / v_sd;
  
  -- Approximate standard normal CDF
  v_percentile := ROUND(100 * (
    0.5 * (1 + SIGN(v_z) * SQRT(1 - EXP(-2 * v_z * v_z / 3.14159)))
  ));
  
  RETURN GREATEST(1, LEAST(99, v_percentile));
END;
$$ LANGUAGE plpgsql;
```
