## TypeScript Percentile Lookup

```typescript
// Percentile-to-RIT lookup tables
interface PercentileLookup {
  [percentile: number]: number; // percentile -> RIT
}

interface GradePercentiles {
  fall: PercentileLookup;
  winter: PercentileLookup;
  spring: PercentileLookup;
}

// Example for Reading, Grade 4, Student
const READING_G4_STUDENT: GradePercentiles = {
  fall: { 5: 166, 10: 173, 15: 177, 20: 180, 25: 183, 50: 196, 75: 208, 90: 219, 95: 226 },
  winter: { 5: 170, 10: 176, 15: 180, 20: 183, 25: 186, 50: 199, 75: 211, 90: 222, 95: 229 },
  spring: { 5: 173, 10: 179, 15: 183, 20: 187, 25: 190, 50: 202, 75: 214, 90: 225, 95: 231 }
};

// Example for Language Usage, Grade 4, Student
const LANGUAGE_USAGE_G4_STUDENT: GradePercentiles = {
  fall: { 5: 167, 10: 173, 15: 177, 20: 180, 25: 183, 50: 195, 75: 206, 90: 216, 95: 222 },
  winter: { 5: 171, 10: 177, 15: 181, 20: 184, 25: 186, 50: 198, 75: 209, 90: 219, 95: 225 },
  spring: { 5: 175, 10: 180, 15: 184, 20: 187, 25: 190, 50: 201, 75: 212, 90: 222, 95: 228 }
};

// Interpolate percentile from RIT
function getPercentileFromRIT(
  rit: number,
  lookup: PercentileLookup
): number {
  const percentiles = Object.keys(lookup).map(Number).sort((a, b) => a - b);
  
  for (let i = 0; i < percentiles.length; i++) {
    const pct = percentiles[i];
    const ritAtPct = lookup[pct];
    
    if (rit <= ritAtPct) {
      if (i === 0) return pct;
      
      // Interpolate
      const prevPct = percentiles[i - 1];
      const prevRit = lookup[prevPct];
      const ratio = (rit - prevRit) / (ritAtPct - prevRit);
      return Math.round(prevPct + ratio * (pct - prevPct));
    }
  }
  
  return 99; // Above 95th percentile
}
```

## SQL Percentile Table

```sql
CREATE TABLE map_achievement_percentiles_2025 (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  level TEXT NOT NULL CHECK (level IN ('student', 'school')),
  subject TEXT NOT NULL,
  grade INTEGER NOT NULL,
  season TEXT NOT NULL,
  percentile INTEGER NOT NULL CHECK (percentile BETWEEN 1 AND 99),
  rit_score INTEGER NOT NULL,
  
  UNIQUE(level, subject, grade, season, percentile)
);

CREATE INDEX idx_pct_lookup ON map_achievement_percentiles_2025(subject, grade, season, level, rit_score);

-- Quick percentile lookup query
SELECT percentile 
FROM map_achievement_percentiles_2025
WHERE subject = 'Reading' 
  AND grade = 4 
  AND season = 'spring' 
  AND level = 'student'
  AND rit_score <= 202
ORDER BY rit_score DESC
LIMIT 1;
```
